## ===================================================== ##
## Commands for package.json
## ===================================================== ##
To run the package.json scripts we always call:

  npm run <script_name>
  npm run webpack




#----------------------------#
# Style and CSS loaders
#----------------------------#
npm install --save-dev style-loader css-loader


#----------------------------#
# Image loaders
#----------------------------#
npm install --save-dev file-loader


#----------------------------#
# Data loaders
#----------------------------#
npm install --save-dev csv-loader xml-loader


#----------------------------#
# HtmlWebpackPlugin
#----------------------------#
The HtmlWebpackPlugin simplifies the creation of HTML files to server your webpack bundles.

npm install --save-dev html-webpack-plugin


#----------------------------#
# CleanWebpackPlugin
#----------------------------#
npm install clean-webpack-plugin --save-dev


#----------------------------#
# Webpack Dev Server
#----------------------------#
npm install --save-dev webpack-dev-server
